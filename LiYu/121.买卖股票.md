121.买卖股票 

解法1 枚举。。

class Solution {

  public int maxProfit(int[] prices) {

 

​    if(prices.length == 0){

​      return 0;

​    }

​    int max = 0; //只要知道最高利润和最低股价

​    int lowPrice = prices[0];

​    for(int i = 1; i < prices.length ; i++){

​       max = Math.max(prices[i]-lowPrice,max); 每次判断一下金额是否最高，股价是否最低。

​       lowPrice = Math.min(lowPrice,prices[i]);

​    }

​    return max;

  }

}
