class Solution {

  public int rob(int[] nums) {

​    if(nums.length == 0) return 0;

​    int[] money = new int[nums.length+1]; 

​    //起始条件

​    money[0] = 0;

​    money[1] = nums[0];

​    for(int i = 2; i <= nums.length; i++){

​      //上一家的值还是 上上家偷上一家的值

​      money[i] = Math.max(money[i - 1], money[i - 2] + nums[i - 1]);

​    }

​    return money[nums.length];

  }

}
